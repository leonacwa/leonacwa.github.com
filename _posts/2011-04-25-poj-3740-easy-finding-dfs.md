#include &lt;stdio.h&gt;  
#include &lt;string.h&gt;  

#define M 512  
#define N 512  

int mm[M][N];  
int m, n, can;  
int sum[N];  

int match()  
{  
    int i;  
    for (i = 0; i &lt; n; i++)  
        if (sum[i] != 1)  
            return 0;  
    return 1;  
}  
void dfs(int x)  
{  
    int i;    
    for (i = 0; i &lt; n; i++) /// 剪枝  
        if (sum[i] &gt; 1){  
            return;  
        }  
    
    if (mm[x][n] &gt; 0){  
        for (i = 0; i &lt; n; i++) /// 选取这行  
            sum[i] += mm[x][i];  
        
        if (match()){ /// 答案  
            can = 1;  
            return;  
        }  
        if (x+1 &lt; m)  
            dfs(x+1);  
        if (can &gt; 0)  
            return;  
        for (i = 0; i &lt; n; i++) /// 不取这行  
            sum[i] -= mm[x][i];  
    }  
    if (x+1 &lt; m &amp;&amp; !can)  
        dfs(x+1);  
}  

int main()  
{  
    int  i, j;  
    while (2 == scanf("%d %d", &amp;m, &amp;n)){  
        memset(mm, 0, sizeof(int)*M*N);  
        for (i = 0; i &lt; m; i++){  
            for (j = 0; j &lt; n; j++){  
                scanf("%d", &amp;mm[i][j]);  
                mm[i][n] += mm[i][j];  
            }  
        }  
        
        memset(sum, 0, sizeof(int)*M);  
        
        can = 0;  
        dfs(0);  
        //for (i = 0; i &lt; n; i++) printf("%d ", sum[i]);printf("n");  
        if (can &gt; 0)  
            printf("Yes, I found itn");  
        else   
            printf("It is impossiblen");  
    }  
    return 0;  
}</pre>
