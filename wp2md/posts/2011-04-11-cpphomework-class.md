/*    
定义一个tree类，有成员ages，成员函数grow(int years)对ages加上    
years，age（）显示tree对象的ages的值    
*/    
#include    
using namespace std;    
    
class Tree{    
  public:    
    Tree(int y = 0):_ages(y)    
    {}    
    Tree(const Tree&amp; t)    
    {    
        _ages = t._ages;    
    }    
    int grow(int years = 0)    
    {    
        _ages += years;    
        return _ages;    
    }    
    int age()    
    {    
        cout &lt;&lt; "tree's ages:" &lt;&lt; _ages;    
        return _ages;    
    }    
    ~Tree()    
    {}    
  private:    
    int _ages;    
};    
    
int main()    
{    
    Tree t1, t2, t3(1);    
    
    t1.grow(1);    
    t1.age();    
    cout &lt;&lt; endl;    
    
    t2 = t1;    
    t2.grow(2);    
    t2.age();    
    cout &lt;&lt; endl;    
    
    t3.age();    
    cout &lt;&lt; endl;    
    
    return 0;
